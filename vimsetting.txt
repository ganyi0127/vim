配置Vim和Gvim
~/.vimrc
  execute pathogen#infect()
  syntax on
  filetype plugin indent on
  set ts=2
  set sw=2
  set expandtab
  set number
  map <C-t> :NERDTreeToggle<CR>
  set autoindent
  inoremap ( ()<Left>
  inoremap { {}<Left>
  inoremap [ []<Left>
  inoremap < <><Left>
  inoremap ' ''<Left>
  inoremap " ""<Left>

~/.gvimrc
  set lines=61
  set columns=159
  colo solarized
  set background=dark
  set guifont=Ubuntu\Mono\11
  NERDTreeToggle
